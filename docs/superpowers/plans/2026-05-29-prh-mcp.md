# prh-mcp Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a FastMCP server wrapping the PRH (Finnish Trade Register) OpenData API, exposing `lookup_company` and `search_companies` tools so Claude plugins can retrieve live Finnish company register data.

**Architecture:** Python FastMCP server, same structure as finlex-mcp. Wraps the PRH OpenData REST API at `https://avoindata.prh.fi/bis/v1/`. No auth required. Deployed to `https://prh-mcp.velvoite.eu/mcp`.

**Tech Stack:** Python 3.11+, `mcp[server]`, `httpx`, `fastapi`, `uvicorn`, `pytest`, `pytest-asyncio`, `pytest-httpx`, `pydantic`.

**PRH API docs:** https://avoindata.prh.fi/bis/v1/ — this API is stable and well-documented.

```python
import httpx
from .models import Company, CompanySearchResult, BoardMember

PRH_BASE = "https://avoindata.prh.fi/bis/v1"


def _parse_company(data: dict) -> Company:
    addresses = data.get("addresses", [])
    address_str = None
    municipality = None
    if addresses:
        addr = addresses[0]
        parts = [addr.get("street"), addr.get("postCode"), addr.get("city")]
        address_str = " ".join(p for p in parts if p)
        municipality = addr.get("city")

    return Company(
        name=data.get("name", ""),
        business_id=data.get("businessId", ""),
        status="Active" if not data.get("dissolvedDate") else "Dissolved",
        registered=data.get("registrationDate"),
        address=address_str,
        municipality=municipality,
        company_form=data.get("companyForm") or (
            data.get("companyForms", [{}])[0].get("type") if data.get("companyForms") else None
        ),
    )


class PRHClient:
    async def lookup_company(self, business_id: str) -> Company:
        async with httpx.AsyncClient(timeout=10.0) as http:
            response = await http.get(f"{PRH_BASE}/{business_id}")
            if response.status_code == 404:
                raise ValueError(f"Company not found: {business_id}")
            response.raise_for_status()
            return _parse_company(response.json())

    async def search_companies(self, name: str, city: str | None = None) -> CompanySearchResult:
        params: dict = {"name": name, "maxResults": 10}
        if city:
            params["city"] = city
        async with httpx.AsyncClient(timeout=10.0) as http:
            response = await http.get(PRH_BASE, params=params)
            response.raise_for_status()
            data = response.json()
            results = [_parse_company(r) for r in data.get("results", [])]
            return CompanySearchResult(
                results=results,
                total=data.get("totalResults", len(results)),
            )
```

- [ ] **Step 2: Write `prh-mcp/src/prh_mcp/server.py`**

```python
from mcp.server.fastmcp import FastMCP
from .client import PRHClient

_client = PRHClient()


def create_server() -> FastMCP:
    mcp = FastMCP(
        "prh",
        instructions=(
            "Provides live Finnish company register data from PRH (Finnish Trade Register). "
            "Use lookup_company with a business ID (Y-tunnus, e.g. '0112038-9') for full details. "
            "Use search_companies to find a company by name when you don't have the business ID."
        ),
    )

    @mcp.tool()
    async def lookup_company(business_id: str) -> dict:
        """
        Look up a Finnish company by business ID (Y-tunnus).
        Returns name, status (Active/Dissolved), address, company form, registration date.

        Args:
            business_id: Finnish business ID in format '1234567-8' (Y-tunnus).

        Returns:
            Company details dict. Raises ValueError if not found.
        """
        company = await _client.lookup_company(business_id)
        return company.model_dump()

    @mcp.tool()
    async def search_companies(name: str, city: str | None = None) -> dict:
        """
        Search Finnish companies by name in the Trade Register.
        Use to find a business ID when you only know the company name.

        Args:
            name: Company name or partial name to search.
            city: Optional city to narrow results (e.g. "Helsinki", "Tampere").

        Returns:
            Dict with 'results' list and 'total' count. Each result has name, business_id, status.
        """
        result = await _client.search_companies(name, city=city)
        return result.model_dump()

    return mcp


def main():
    import uvicorn
    import os
    server = create_server()
    app = server.get_asgi_app()
    uvicorn.run(app, host="0.0.0.0", port=int(os.environ.get("PORT", "8081")))


if __name__ == "__main__":
    main()
```
